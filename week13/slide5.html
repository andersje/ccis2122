    ACTION PROTOCOL SRCADDR SRCPORT DESTADDR DESTPORT ( RULE OPTIONS; )

    alert tcp any any -> LOCALIP 22 (msg: 'attempts SSH connect'; sid: 123456; )

    alert tcp any any -> any any (msg: 'net traffic';)
